\choice $\frac{89}{95}$

\uplevel{Questions~\ref{FirstDice}--\ref{LastDice}
involve the experiment of rolling two regular
six-sided dice.}

\question\label{FirstDice}
How many outcomes are in the sample space of the experiment?\\
\begin{oneparchoices}
\choice $6$ % one die
\choice $11$ % possible sums of two dice
\choice $12$ % add rather than multiply
\choice $18$
\correctchoice $36$
\end{oneparchoices}

\question What is the probability that both dice show six?\\
\begin{oneparchoices}
\choice $\frac{1}{30}$ % replaced, although impossible
\correctchoice $\frac{1}{36}$
\choice $\frac{1}{12}$ % took 12 to be size of sample space
\choice $\frac{1}{6}$
\choice $\frac{1}{5}$ % number from stem
\end{oneparchoices}

\question
What is the probability that at least one die shows six?\\
\begin{oneparchoices}
\choice $\frac{1}{36}$
\choice $\frac{1}{6}$
\choice $\frac{1}{5}$ % number from stem
\correctchoice $\frac{11}{36}$
\choice $\frac{25}{36}$ % complement
\end{oneparchoices}

\question\label{LastDice}
What is the probability that six is the sum of the numbers
showing on the dice?
\begin{oneparchoices}
\choice $\frac{1}{36}$
\choice $\frac{1}{12}$ % 6 + 6
\correctchoice $\frac{5}{36}$
\choice $\frac{5}{31}$ % odds
\choice $\frac{1}{6}$
\end{oneparchoices}


\uplevel{Questions~\ref{FirstPresident}--\ref{LastPresident}
deal with the following situation.
You need to select colors to paint your bedroom and bathroom.
You have narrowed the selection down to four colors, namely,
aqua sparkle, battleship grey, coral, and deep carmine.
However, if aqua sparkle is used in one room, deep carmine orange should not
be used in the other, since the two colors don't match.}

\question\label{FirstPresident}
In how many ways can the two colors be selected?\\
\begin{oneparchoices}
\choice $5$ % number of combinations
\correctchoice $10$
\choice $11$ % only eliminated one A/D
\choice $12$ % 3 by 4
\choice $14$ % replaced
\end{oneparchoices}

\question
What is the probability that battleship will be chosen
for one or the other room\\
\begin{oneparchoices}
\choice $\frac{2}{5}$ % counted second branch as 1
\correctchoice $\frac{3}{5}$
\choice $\frac{1}{2}$ % heuristic, or didn't eliminate A/D
\choice $\frac{7}{16}$ % replaced and didn't eliminate A/D
\choice $\frac{5}{16}$ % replaced 
\end{oneparchoices}
