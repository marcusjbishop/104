\question A poker hand is called a {\em spade flush}
if all five cards have suit $\spadesuit$. What is
the number of possible spade flushes?\\
\begin{oneparchoices}
\correctchoice $1287$
\choice $154,440$ % P(13,5)
\end{oneparchoices}

\question What is $\ncr{12}{5}$?\\
\begin{oneparchoices}
\choice $60$ % 12 * 5
\correctchoice $792$
\choice $95,040$ % P(12,5)
\end{oneparchoices}

\question How many subsets of size four does the set
$\left\{a,b,c,d,e,f,g\right\}$ have?\\
\begin{oneparchoices}
\choice $15$ %C(6,4)
\correctchoice $35$
\choice $128$ % 2^7
\choice $360$ % P(6,4)
\choice $840$ % P(7,4)
\end{oneparchoices}
