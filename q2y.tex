\question Suppose $E$ and $F$ are events.
Suppose also that $P\left(E\right)=0.65$,
$P\left(\text{$E$ and $F$}\right)=0.45$
and $P\left(\text{$E$ or $F$}\right)=0.95$. What
is $P\left(F\right)$?\\
\begin{oneparchoices}
\choice $0.15$ % .65+.45-.95
\choice $0.20$ % E\F
\choice $0.30$ % F\E
\choice $0.50$ % E\F + F\E
\correctchoice $0.75$
\end{oneparchoices}

\question An octahedral die has eight sides labeled $1,2,\ldots,8$.
If an experiment consists of rolling four such dice, how
many outcomes are in the sample space of the experiment?\\
\begin{oneparchoices}
\choice $8$ % faces on one die
\choice $16$
\choice $32$ % number of faces
\choice $64$ % two instead of four dice
\correctchoice $4096$
\end{oneparchoices}

\question Ames Middle School has
$148$ students in band and $35$
students in both band and chorus.
If $723$ of the $1046$ students at Ames High
are in neither band nor chorus, how many students
are in chorus?\\
\begin{oneparchoices}
\choice $113$ % just band
\choice $175$ % just chorus
\correctchoice $210$
\choice $323$ % number in music
\choice $610$ % took 723 to be number in music
\end{oneparchoices}

\uplevel{Questions~\ref{FirstDice}--\ref{LastDice}
involve the experiment of rolling two regular
six-sided dice.}

\question\label{FirstDice}
How many outcomes are in the sample space of the experiment?\\
\begin{oneparchoices}
\choice $6$ % one die
\choice $11$ % possible sums of two dice
\choice $12$ % add rather than multiply
\choice $18$
\correctchoice $36$
\end{oneparchoices}

\question What is the probability that at least one die shows three?\\
\begin{oneparchoices}
\choice $\frac{1}{36}$
\choice $\frac{1}{6}$
\choice $\frac{1}{5}$ % number from stem
\correctchoice $\frac{11}{36}$
\choice $\frac{25}{36}$ % complement
\end{oneparchoices}

\question\label{LastDice}
What is the probability that both dice show three?\\
\begin{oneparchoices}
\choice $\frac{1}{30}$ % replaced, although impossible
\correctchoice $\frac{1}{36}$
\choice $\frac{1}{12}$ % took 12 to be size of sample space
\choice $\frac{1}{6}$
\choice $\frac{1}{5}$ % number from stem
\end{oneparchoices}

\uplevel{Questions~\ref{FirstPresident}--\ref{LastPresident}
deal with the student council.
Alex, Brooke, Colby, and Denise comprise the student
council. They want to select a president and
a vice president from among themselves. However,
Alex and Denise cannot serve together in these roles
because they are married to each other.}

\question\label{FirstPresident}
How many possible choices of president and vice president are there?\\
\begin{oneparchoices}
\choice $5$ % number of combinations
\correctchoice $10$
\choice $11$ % only eliminated one A/D
\choice $12$ % 3 by 4
\choice $14$ % replaced
\end{oneparchoices}

\question
In how many choices does Denise appear as either president
or vice president?\\
\begin{oneparchoices}
\correctchoice $4$
\choice $5$ % heuristic, half the choices
\choice $6$ % didn't eliminate A/D
\choice $7$
\choice $8$
\end{oneparchoices}

\question
What is the probability that Colby will be chosen
as either president or vice president?\\
\begin{oneparchoices}
\choice $\frac{2}{5}$ % counted second branch as 1
\correctchoice $\frac{3}{5}$
\choice $\frac{1}{2}$ % heuristic, or didn't eliminate A/D
\choice $\frac{7}{16}$ % replaced and didn't eliminate A/D
\choice $\frac{5}{16}$ % replaced 
\end{oneparchoices}

\question\label{LastPresident}
What is the probability that Alex will be selected
as vice president?\\
\begin{oneparchoices}
\correctchoice $\frac{1}{5}$
\choice $\frac{1}{4}$ % replaced and didn't eliminate A/D
\choice $\frac{3}{10}$ % didn't eliminate A/D
\choice $\frac{3}{16}$ % replaced 
\choice $\frac{1}{2}$ % heuristic
\end{oneparchoices}
